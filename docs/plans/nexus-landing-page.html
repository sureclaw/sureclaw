Deploy in minutes

Get started with our intuitive API. Deploy your first AI employee with just a few lines of code.

deploy.js 
import { Nexus } from '@nexus/sdk';

// Initialize client with API key
const nexus = new Nexus({
  apiKey: process.env.NEXUS_API_KEY,
  secure: true
});

// Deploy a new AI employee
const employee = await nexus.employees.create({
  name: 'atlas-scout',
  role: 'research-analyst',
  capabilities: ['data-analysis', 'report-generation'],
  autoStart: true
});

// Assign a task
await employee.assignTask({
  type: 'analyze',
  data: marketData,
  priority: 'high'
});

console.log(`employee ${employee.id} active`);
